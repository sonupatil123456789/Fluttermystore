                              int _currentStep = 0;

  int get currentStep => _currentStep;

  tapped(int step) {
    _currentStep = step;
    notifyListeners();
  }

  continuedStep() {
    _currentStep < 3 ? _currentStep += 1 : null;
    notifyListeners();
  }

  cancelStep() {
    _currentStep > 0 ? _currentStep -= 1 : null;
    notifyListeners();
  }
